dim_dealer
-
dealer_id int PK
country_name varchar(100)
location varchar(100)
dealer_name varchar(100)
dealer_contact varchar(100)



dim_branch
-
branch_id int PK
name varchar(100)
address varchar(100)
country varchar(100)



dim_date
-
date_id int PK
date date



fact_revenue
-
dealer_id int PK FK >- dim_dealer.dealer_id
date_id int PK FK >- dim_date.date_id
branch_id int PK FK >- dim_branch.branch_id
units_sold int
revenue decimal(8,2)


dim_product
-
product_id int PK
model_id int
product_name varchar(100)
variant_name varchar(100)
fuel_type varchar(100)



fact_supply
-
product_id int PK FK >- dim_product.product_id
branch_id int PK FK >- dim_branch.branch_id
date_id int PK FK >- dim_date.date_id
units_supplied int
